import matplotlib.pyplot as plt
from scipy import stats

def get_text(pv):
    p_text = ''
    if pv<0.001:
        p_text = '***'
    elif 0.001 < pv < 0.01:
        p_text = '**'
    elif 0.01 < pv < 0.1:
        p_text = '*'
    else:
        p_text = '#'
    return p_text

def plot_ablation(G):
    plt.figure(figsize=(15,10))
    
    labels = ["TuSDC-S", "TuSDC-SC1", "TuSDC-SC2", "TuSDC-SM", "TuSDC", ]

    bplot1 = ax1.boxplot(data, notch=True, vert=True, patch_artist=True)
    ax1.set_ylabel('Hits@10')
    ax1.set_xticklabels(labels, rotation=30)
    pv1=stats.ttest_ind(G.TuSDC_SC1[0], G.TuSDC_S[0], alternative='two-sided').pvalue
    pv2 = stats.ttest_ind(G.TuSDC_SC2[0], G.TuSDC_SC1[0], alternative='two-sided').pvalue
    pv3 = stats.ttest_ind(G.TuSDC_SM[0], G.TuSDC_S[0], alternative='two-sided').pvalue
    pv4 = stats.ttest_ind(G.TuSDC[0], G.TuSDC_SC2[0], alternative='two-sided').pvalue
    pv5 = stats.ttest_ind(G.TuSDC[0], G.TuSDC_SM[0], alternative='two-sided').pvalue
